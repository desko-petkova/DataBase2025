--- Форсирано прекъсване на връзката със сървъра за да се изтрие база данни ----
--------------------------------------------------------------------------------
	ALTER DATABASE [databaseName] SET SINGLE_USER WITH ROLLBACK IMMEDIATE;
	DROP DATABASE [databaseName];
	
--------------------------------------------------------------------------------
--- 		бърз начин да видиш структурата на таблицата 		    ----
--------------------------------------------------------------------------------

	EXEC sp_help '[column_name]';

--------------------------------------------------------------------------------
--- структурата на таблицата [column_name] – колони, типове, дължина,       ----
	 				     дали позволява нулеви стойности
--------------------------------------------------------------------------------

	SELECT COLUMN_NAME, DATA_TYPE, CHARACTER_MAXIMUM_LENGTH, IS_NULLABLE
	FROM INFORMATION_SCHEMA.COLUMNS
	WHERE TABLE_NAME = '[column_name]';
