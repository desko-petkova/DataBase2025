CREATE TABLE Teachers
(
    teacher_id INT PRIMARY KEY,
    name NVARCHAR(100),
    manager_id INT,
    
    -- Външен ключ: manager_id => teachers.teacher_id
    CONSTRAINT fk_teacher_manager
    FOREIGN KEY (manager_id) 
    REFERENCES teachers(teacher_id)
);
