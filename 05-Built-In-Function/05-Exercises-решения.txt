SELECT FirstName, LastName
FROM Employees
WHERE LEFT(FirstName, 2) = 'SA'
--
SELECT FirstName, LastName
FROM Employees
WHERE SUBSTRING(FirstName, 1, 2) = 'Sa'
-- 02 ---
SELECT FirstName, LastName
FROM Employees
WHERE CHARINDEX('EI', LastName, 1) != 0
---
SELECT FirstName, LastName
FROM Employees
WHERE LastName LIKE '%ei%'
-- 03 ---
SELECT FirstName, DepartmentID, HireDate
FROM Employees
WHERE (DepartmentID = 3 OR DepartmentID = 10) AND
YEAR(HireDate) >= 1995 AND YEAR(HireDate) <= 2005
--
SELECT FirstName, DepartmentID, HireDate
FROM Employees
WHERE DepartmentID IN (3,10) AND YEAR(HireDate) BETWEEN 1995 AND 2005 
--- 04 --
SELECT FirstName, JobTitle
FROM Employees
WHERE CHARINDEX('engineer', JobTitle, 1) = 0
-- 05 --
SELECT [Name], TownID
FROM Towns
WHERE LEN([Name]) = 5 OR  LEN([Name]) = 6
ORDER BY [Name] ASC -- DESC
-- 06 ---
SELECT TownID,[Name]
FROM Towns
WHERE LEFT([Name],1)  IN ('M', 'K','B','E')
ORDER BY [Name] DESC
-- 07 --
SELECT TownID,[Name]
FROM Towns
WHERE LEFT([Name],1) NOT IN ('R', 'D','B')
ORDER BY [Name] ASC

USE Geography
SELECT PeakName, RiverName,
LOWER(CONCAT(PeakName,RiverName)) AS Mix
FROM Peaks, Rivers
WHERE LEFT(RiverName, 1) = RIGHT(PeakName,1)
ORDER BY Mix ASC
